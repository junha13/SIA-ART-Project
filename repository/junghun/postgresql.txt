슈퍼계정 연동

-- ① 슈퍼계정 6개 생성 (LOGIN + SUPERUSER)
DO $$
DECLARE u text;
BEGIN
  FOREACH u IN ARRAY ARRAY['admin1','admin2','admin3','admin4','admin5','admin6']
  LOOP
    EXECUTE format($f$
      DO $x$ BEGIN
        IF NOT EXISTS (SELECT 1 FROM pg_roles WHERE rolname = '%1$s') THEN
          CREATE ROLE "%1$s" WITH
            LOGIN
            SUPERUSER
            INHERIT
            BYPASSRLS
            CONNECTION LIMIT -1
            PASSWORD '%1$s_strong_pw!'; -- 👉 비밀번호 바꾸기
        ELSE
          ALTER ROLE "%1$s" WITH
            LOGIN SUPERUSER INHERIT BYPASSRLS CONNECTION LIMIT -1
            PASSWORD '%1$s_strong_pw!'; -- 👉 비밀번호 바꾸기
        END IF;
      END $x$;
    $f$, u);
  END LOOP;
END$$;

-- ② 확인
\du
